%% Default line width
\pgfplotsset{
  every axis plot/.append style={line width=1pt,},
}
\pgfplotsset{/pgfplots/group/every plot/.append style = {
   	axis line style = semithick,
   	every tick/.style={black,semithick,},
  },
}
\pgfplotsset{every axis/.append style={
    ymajorgrids,
    xmajorgrids,
    grid style={dashed, lightgray,semithick},
   },
}

% Multiplot settings
\pgfplotsset{/pgfplots/group/every plot/.append style = {
    height = 5.5cm, width = 5.75cm,
    ylabel near ticks,
  }
}

% text in legend entry
% source: https://tex.stackexchange.com/a/204519
\pgfplotsset{
    legend image with text/.style={
        legend image code/.code={%
            \node[anchor=center] at (0.3cm,0cm) {#1};
        }
    },
}

% contour legend entry
\pgfplotsset{
	/pgfplots/contour legend/.style={
		legend image code/.code={
			\draw[color=red, line width=0.8pt] (0.3cm,0.05cm) circle (0.2cm);
			\draw[color=orange, line width=0.8pt] (0.3cm,0.05cm) circle (0.125cm);
			\draw[color=yellow, line width=0.8pt] (0.3cm,0.05cm) circle (0.05cm);
		},
	},
}
